def matrix_multiply(first,second):
    """
    2
    2
    """
    result = []
    temp = 0.0
    for m in range(len(first)):
        result.append([])
        for n in range(len(second[0])):
            temp = 0.0
            for order in range(len(first[0])):
                temp += first[m][order] * second[order][n]
            result[m].append(temp)
    return result

def discriminant(a, b, c):
    """
    a x**2 + b x + c = 0 
     `two real roots`  `one real root`
     `two complex roots` 
    """
    d = b**2 - 4.0*a*c
    if d > 0.0:
        print('two real roots')
    elif d == 0.0:
        print('one real root')
    else:
        print('two complex roots')
